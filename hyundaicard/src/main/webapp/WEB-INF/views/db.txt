테이블 만들기

create table book(
/*bigint unsigned 매우큰 양의 정수 0 ~ 18,446,744,073,709,551,615
not null 값이 반드시 있어야 함 
auto_increment 자동으로 1씩 증가
*/
book_id bigint unsigned not null auto_increment primary key,
/*html input 네이밍과 상동*/
title varchar(200) not null,
category varchar(200) not null default '',
price int not null,
insert_date datetime not null default now()
) 
collate='utf8mb4_general_ci';
/*
utf8mb4 유니코드 문자 집합, 이모지 포함 모든 유니코드 문자 지원
general_ci 대소문자 구분 안함
*/

select * from book;
